11 auto split(const std::string& s, const std::string& delim)
12 -> std::vector<std::string> {
13 size_t pos_start{};
14 size_t pos_end{};
15 std::vector<std::string> ret;
16
17 while((pos_end = s.find(delim, pos_start)) != std::string::npos) {
18 auto token = s.substr(pos_start, pos_end - pos_start);
19 pos_start = pos_end + delim.size();
20 ret.emplace_back(token);
21 }
22
23 ret.emplace_back(s.substr(pos_start));
24 return ret;
25 }

27 auto parse_args(int argc, char** argv)
28 -> std::optional<std::unordered_map<std::string, std::string>> {
29 auto opts = std::unordered_map<std::string, std::string>();
30 auto args = cbdc::config::get_args(argc, argv);
31 for(size_t i = 1; i < args.size(); i++) {
32 const auto& arg = args[i];
33 auto arr = split(arg, "--");
34 if(arr.size() != 2 || !arr[0].empty() || arr[1].empty()) {
35 return std::nullopt;
36 }
37
38 auto elems = split(arr[1], "=");
39 if(elems.size() != 2) {
40 return std::nullopt;
41 }
42
43 opts.emplace(elems[0], elems[1]);
44 }
45 return opts;
46 }

49 read_endpoints(const std::unordered_map<std::string, std::string>& opts,
50 const std::string& component_name)
51 -> std::optional<std::vector<network::endpoint_t>> {
52 auto ret = std::vector<network::endpoint_t>();
53
54 auto count_key = component_name + "_count";
55 auto it = opts.find(count_key);
56 if(it == opts.end()) {
57 return std::nullopt;
58 }
59 auto count = std::stoull(it->second);
60
61 for(size_t i = 0; i < count; i++) {
62 auto ep_key = component_name + std::to_string(i) + "_endpoint";
63 it = opts.find(ep_key);
64 if(it == opts.end()) {
65 return std::nullopt;
66 }
67 auto ep = cbdc::config::parse_ip_port(it->second);
68 ret.emplace_back(ep);
69 }
70
71 return ret;
72 }

194 auto put_row(const std::shared_ptr<broker::interface>& broker,
195 broker::key_type key,
196 broker::value_type value,
197 const std::function<void(bool)>& result_callback) -> bool {
198 auto begin_res = broker->begin([=](auto begin_ret) {
199 if(!std::holds_alternative<
200 cbdc::parsec::ticket_machine::ticket_number_type>(
201 begin_ret)) {
202 result_callback(false);
203 return;
204 }
205 auto ticket_number
206 = std::get<cbdc::parsec::ticket_machine::ticket_number_type>(
207 begin_ret);
208 auto lock_res = broker->try_lock(
209 ticket_number,
210 key,
211 cbdc::parsec::runtime_locking_shard::lock_type::write,
212 [=](auto try_lock_res) {
213 if(!std::holds_alternative<cbdc::buffer>(try_lock_res)) {
214 result_callback(false);
215 return;
216 }
217 auto commit_res = broker->commit(
218 ticket_number,
219 {{key, value}},
220 [=](auto commit_ret) {
221 if(commit_ret.has_value()) {
222 result_callback(false);
223 return;
224 }
225 auto finish_res = broker->finish(
226 ticket_number,
227 [=](auto finish_ret) {
228 result_callback(!finish_ret.has_value());
229 });
230 if(!finish_res) {
231 result_callback(false);
232 return;
233 }
234 });
235 if(!commit_res) {
236 result_callback(false);
237 return;
238 }
239 });
240 if(!lock_res) {
241 result_callback(false);
242 return;
243 }
244 });
245 return begin_res;
246 }
