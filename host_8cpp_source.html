39 auto evm_host::get_account(const evmc::address& addr, bool write) const
40 -> std::optional<evm_account> {
41 m_log->trace(this,
42 "EVM request account:",
43 to_hex(addr),
44 "- Write:",
45 write);
46
47 if(is_precompile(addr)) {
48 // Precompile contract, return empty account
49 m_accessed_addresses.insert(addr);
50 return evm_account{};
51 }
52
53 auto it = m_accounts.find(addr);
54 if(it != m_accounts.end() && (it->second.second || !write)) {
55 return it->second.first;
56 }
57
58 m_log->trace(
59 this,
60 "EVM request account not in cache or wrong lock, getting [",
61 to_hex(addr),
62 "]");
63
64 auto addr_key = make_buffer(addr);
65 auto maybe_v = get_key(addr_key, write);
66 if(!maybe_v.has_value()) {
67 return std::nullopt;
68 }
69
70 m_accessed_addresses.insert(addr);
71 auto& v = maybe_v.value();
72 if(v.size() == 0) {
73 m_accounts[addr] = {std::nullopt, write};
74 return std::nullopt;
75 }
76 auto maybe_acc = from_buffer<evm_account>(v);
77 assert(maybe_acc.has_value());
78 auto& acc = maybe_acc.value();
79 m_accounts[addr] = {acc, write};
80 return acc;
81 }

104 auto evm_host::set_storage(const evmc::address& addr,
105 const evmc::bytes32& key,
106 const evmc::bytes32& value) noexcept
107 -> evmc_storage_status {
108 const auto* log_str = "evm_set_storage";
109 m_log->trace(log_str, to_hex(addr), to_hex(key), to_hex(value));
110
111 auto ret_val = std::optional<evmc_storage_status>();
112 auto maybe_acc = get_account(addr, false);
113 if(!maybe_acc.has_value()) {
114 if(!m_is_readonly_run) {
115 maybe_acc = get_account(addr, true);
116 assert(!maybe_acc.has_value());
117 }
118 maybe_acc = evm_account();
119 ret_val = EVMC_STORAGE_ADDED;
120 maybe_acc.value().m_modified.insert(key);
121 m_accounts[addr] = {maybe_acc.value(), !m_is_readonly_run};
122 }
123 auto& acc = maybe_acc.value();
124
125 auto maybe_storage
126 = get_account_storage(addr, key, !m_is_readonly_run);
127 auto prev_value = maybe_storage.value_or(evmc::bytes32{});
128
129 auto modified = acc.m_modified.find(key) != acc.m_modified.end();
130 if(!ret_val.has_value()) {
131 if(prev_value == value || modified) {
132 // NOTE: both unchanged value and modifying previously
133 // added/modified value cases belong in the same group
134 // related to minimal gas cost of only accessing warm storage.
135 // See evmc.h for more details.
136 ret_val = EVMC_STORAGE_ASSIGNED;
137 } else if(evmc::is_zero(value) && !modified) {
138 ret_val = EVMC_STORAGE_DELETED;
139 } else {
140 ret_val = EVMC_STORAGE_MODIFIED;
141 acc.m_modified.insert(key);
142 }
143 }
144 m_account_storage[addr][key] = {value, !m_is_readonly_run};
145 m_accounts[addr].first = acc;
146 assert(ret_val.has_value());
147 return ret_val.value();
148 }

175 auto evm_host::get_code_hash(const evmc::address& addr) const noexcept
176 -> evmc::bytes32 {
177 const auto* log_str = "evm_get_code_hash";
178 m_log->trace(log_str, to_hex(addr));
179
180 auto maybe_code = get_account_code(addr, false);
181 if(!maybe_code.has_value()) {
182 return {};
183 }
184 auto& code = maybe_code.value();
185 auto sha = CSHA256();
186 sha.Write(code.data(), code.size());
187 auto ret = evmc::bytes32();
188 sha.Finalize(&ret.bytes[0]);
189 return ret;

192 auto evm_host::copy_code(const evmc::address& addr,
193 size_t code_offset,
194 uint8_t* buffer_data,
195 size_t buffer_size) const noexcept -> size_t {
196 const auto* log_str = "evm_copy_code";
197 m_log->trace(log_str, to_hex(addr), code_offset);
198
199 auto maybe_code = get_account_code(addr, false);
200 if(!maybe_code.has_value()) {
201 return 0;
202 }
203
204 const auto& code = maybe_code.value();
205
206 if(code_offset >= code.size()) {
207 return 0;
208 }
209
210 const auto n = std::min(buffer_size, code.size() - code_offset);
211 if(n > 0) {
212 std::copy_n(&code[code_offset], n, buffer_data);
213 }
214 return n;
215 }

226 auto evm_host::create(const evmc_message& msg) noexcept -> evmc::Result {
227 auto maybe_sender_acc = get_account(msg.sender, false);
228 assert(maybe_sender_acc.has_value());
229 auto& sender_acc = maybe_sender_acc.value();
230
231 auto new_addr = evmc::address();
232 if(msg.kind == EVMC_CREATE) {
233 new_addr = contract_address(msg.sender, sender_acc.m_nonce);
234 } else {
235 auto bytecode_hash
236 = cbdc::keccak_data(msg.input_data, msg.input_size);
237 new_addr = contract_address2(msg.sender,
238 msg.create2_salt,
239 bytecode_hash);
240 }
241
242 // Transfer endowment to deployed contract account
243 if(!evmc::is_zero(msg.value)) {
244 transfer(msg.sender, new_addr, msg.value);
245 }
246
247 if(msg.depth == 0) {
248 m_receipt.m_create_address = new_addr;
249 }
250
251 auto call_msg = evmc_message();
252 call_msg.depth = msg.depth;
253 call_msg.sender = msg.sender;
254 call_msg.value = msg.value;
255 call_msg.recipient = new_addr;
256 call_msg.kind = EVMC_CALL;
257 // TODO: do we need to deduct some gas for contract creation
258 // here?
259 call_msg.gas = msg.gas;
260
261 auto res = execute(call_msg, msg.input_data, msg.input_size);
262
263 if(res.status_code == EVMC_SUCCESS) {
264 // The VM does not populate res.create_address, and it must
265 // set to the address of the new contract.
266 // If left as 0x0, deployment of contracts by another contract
267 // (i.e. depth>0) is likely to fail (e.g.
268 // UniswapV3Factory.createPool())
269 res.create_address = new_addr;
270
271 auto maybe_acc = get_account(new_addr, !m_is_readonly_run);
272 if(!maybe_acc.has_value()) {
273 maybe_acc = evm_account();
274 }
275 auto& acc = maybe_acc.value();
276 m_accounts[new_addr] = {acc, !m_is_readonly_run};
277
278 auto maybe_code = get_account_code(new_addr, !m_is_readonly_run);
279 if(!maybe_code.has_value()) {
280 maybe_code = evm_account_code();
281 }
282 auto& code = maybe_code.value();
283 code.resize(res.output_size);
284 std::memcpy(code.data(), res.output_data, res.output_size);
285 m_account_code[new_addr] = {code, !m_is_readonly_run};
286 }
287
288 if(msg.depth == 0) {
289 m_receipt.m_output_data.resize(res.output_size);
290 std::memcpy(m_receipt.m_output_data.data(),
291 res.output_data,
292 res.output_size);
293 m_receipt.m_success
294 = res.status_code == evmc_status_code::EVMC_SUCCESS;
295 }
296
297 return res;
298 }

300 auto evm_host::call(const evmc_message& msg) noexcept -> evmc::Result {
301 if(msg.kind == EVMC_CREATE2 || msg.kind == EVMC_CREATE) {
302 return create(msg);
303 }
304
305 // Transfer message value from sender account to recipient
306 bool is_native_value_transfer{false};
307 if(!evmc::is_zero(msg.value) && msg.kind == EVMC_CALL) {
308 is_native_value_transfer = true;
309 // TODO: do DELETEGATECALL and CALLCODE transfer value as
310 // well?
311 transfer(msg.sender, msg.recipient, msg.value);
312 }
313
314 auto code_addr
315 = msg.kind == EVMC_DELEGATECALL || msg.kind == EVMC_CALLCODE
316 ? msg.code_address
317 : msg.recipient;
318
319 const auto code_size = get_code_size(code_addr);
320 if(code_size == 0) {
321 // TODO: deduct simple send fixed gas amount
322 const auto gas_refund = 0;
323 auto res = evmc::make_result(evmc_status_code::EVMC_SUCCESS,
324 msg.gas,
325 gas_refund,
326 nullptr,
327 0);
328
329 // TODO: Is it possible to have a case where this is not a native
330 // value transfer (i.e. is else-block needed here)?
331 if(is_native_value_transfer) {
332 m_receipt.m_success = true;
333 }
334
335 return evmc::Result(res);
336 }
337
338 auto code_buf = std::vector<uint8_t>(code_size);
339 [[maybe_unused]] auto n
340 = copy_code(code_addr, 0, code_buf.data(), code_buf.size());
341 assert(n == code_size);
342
343 auto inp = cbdc::buffer();
344 inp.append(msg.input_data, msg.input_size);
345 m_log->trace("EVM call:",
346 to_hex(code_addr),
347 msg.kind,
348 msg.flags,
349 msg.depth,
350 inp.to_hex());
351
352 auto res = execute(msg, code_buf.data(), code_buf.size());
353
354 if(msg.depth == 0) {
355 // TODO: refactor branch into call epilog method
356 m_receipt.m_output_data.resize(res.output_size);
357 std::memcpy(m_receipt.m_output_data.data(),
358 res.output_data,
359 res.output_size);
360 m_receipt.m_success
361 = res.status_code == evmc_status_code::EVMC_SUCCESS;
362 }
363
364 return res;
365 }

473 auto evm_host::get_state_updates() const
474 -> runtime_locking_shard::state_update_type {
475 auto ret = runtime_locking_shard::state_update_type();
476 for(auto& [addr, acc_data] : m_accounts) {
477 auto& [acc, write] = acc_data;
478 if(!acc.has_value() || !write) {
479 continue;
480 }
481 auto key = make_buffer(addr);
482 auto val = cbdc::buffer();
483 if(!acc->m_destruct) {
484 val = make_buffer(*acc);
485 }
486 ret[key] = val;
487 }
488
489 for(auto& [addr, acc_code] : m_account_code) {
490 auto& [code, write] = acc_code;
491 if(!code.has_value() || !write) {
492 continue;
493 }
494 auto key = make_buffer(code_key{addr});
495 auto val = make_buffer(*code);
496 ret[key] = val;
497 }
498
499 for(auto& [addr, acc_storage] : m_account_storage) {
500 for(auto& [k, elem] : acc_storage) {
501 auto& [value, write] = elem;
502 if(!value.has_value() || !write) {
503 continue;
504 }
505 auto key = make_buffer(storage_key{addr, k});
506 auto val = make_buffer(*value);
507 ret[key] = val;
508 }
509 }
510
511 const auto tx = std::make_shared<evm_tx>(m_tx);
512 auto txid = tx_id(*tx);
513 auto tid = make_buffer(txid);
514 auto r = make_buffer(m_receipt);
515 ret[tid] = r;
516 ret[ticket_number_key()] = tid;
517
518 auto ordered_logs = get_sorted_logs();
519 for(auto& addr_log : ordered_logs) {
520 auto log_idx = evm_log_index();
521 log_idx.m_ticket_number = m_ticket_number;
522 log_idx.m_txid = txid;
523 log_idx.m_logs = addr_log.second;
524 ret[log_index_key(addr_log.first)] = make_buffer(log_idx);
525 }
526 return ret;
527 }

540 void evm_host::transfer(const evmc::address& from,
541 const evmc::address& to,
542 const evmc::uint256be& value) {
543 const auto* log_str = "evm_transfer";
544 m_log->trace(log_str, to_hex(from), to_hex(to));
545 auto maybe_acc = get_account(from, !m_is_readonly_run);
546 assert(maybe_acc.has_value());
547 auto& acc = maybe_acc.value();
548 auto val = value;
549 if(evmc::is_zero(value)) {
550 // Special case: destruct the from account if we're
551 // transfering the entire account balance
552 val = acc.m_balance;
553 acc.m_destruct = true;
554 }
555 acc.m_balance = acc.m_balance - val;
556 m_accounts[from] = {acc, !m_is_readonly_run};
557
558 auto maybe_to_acc = get_account(to, !m_is_readonly_run);
559 if(!maybe_to_acc.has_value()) {
560 // Create the to account if it doesn't exist
561 maybe_to_acc = evm_account();
562 }
563 auto& to_acc = maybe_to_acc.value();
564 to_acc.m_balance = to_acc.m_balance + val;
565 m_accounts[to] = {to_acc, !m_is_readonly_run};
566 }

568 void evm_host::finalize(int64_t gas_left, int64_t gas_used) {
569 if(!m_is_readonly_run) {
570 auto maybe_acc = get_account(m_tx_context.tx_origin, true);
571 assert(maybe_acc.has_value());
572 auto& acc = maybe_acc.value();
573 auto gas_refund = evmc::uint256be(static_cast<uint64_t>(gas_left))
574 * m_tx_context.tx_gas_price;
575 acc.m_balance = acc.m_balance + gas_refund;
576 m_accounts[m_tx_context.tx_origin] = {acc, true};
577 }
578 m_receipt.m_gas_used
579 = evmc::uint256be(static_cast<uint64_t>(gas_used));
580 m_receipt.m_timestamp = static_cast<uint64_t>(
581 std::chrono::duration_cast<std::chrono::seconds>(
582 std::chrono::system_clock::now().time_since_epoch())
583 .count());
584 }

590 auto evm_host::is_precompile(const evmc::address& addr) -> bool {
591 auto addr_copy = addr;
592 constexpr auto precompile_suffix_sz = sizeof(uint16_t);
593 std::memset(
594 &addr_copy.bytes[sizeof(addr_copy.bytes) - precompile_suffix_sz],
595 0,
596 precompile_suffix_sz);
597 return evmc::is_zero(addr_copy)
598 && addr.bytes[sizeof(addr_copy.bytes) - 1] != 0;
599 }

601 auto evm_host::get_account_storage(const evmc::address& addr,
602 const evmc::bytes32& key,
603 bool write) const
604 -> std::optional<evmc::bytes32> {
605 m_log->trace("EVM request account storage:",
606 to_hex(addr),
607 to_hex(key));
608
609 if(is_precompile(addr)) {
610 // Precompile contract, return empty account
611 m_accessed_addresses.insert(addr);
612 return std::nullopt;
613 }
614
615 auto it = m_account_storage.find(addr);
616 if(it != m_account_storage.end()) {
617 auto& m = it->second;
618 auto itt = m.find(key);
619 if(itt != m.end() && (itt->second.second || !write)) {
620 return itt->second.first;
621 }
622 }
623
624 auto elem_key = make_buffer(storage_key{addr, key});
625 auto maybe_v = get_key(elem_key, write);
626 if(!maybe_v.has_value()) {
627 return std::nullopt;
628 }
629
630 m_accessed_addresses.insert(addr);
631 auto& v = maybe_v.value();
632 if(v.size() == 0) {
633 m_account_storage[addr][key] = {std::nullopt, write};
634 return std::nullopt;
635 }
636 auto maybe_data = from_buffer<evmc::bytes32>(v);
637 assert(maybe_data.has_value());
638 auto& data = maybe_data.value();
639 m_account_storage[addr][key] = {data, write};
640 return data;
641 }

643 auto evm_host::get_account_code(const evmc::address& addr,
644 bool write) const
645 -> std::optional<evm_account_code> {
646 m_log->trace("EVM request account code:", to_hex(addr));
647
648 if(is_precompile(addr)) {
649 // Precompile contract, return empty account
650 m_accessed_addresses.insert(addr);
651 return std::nullopt;
652 }
653
654 auto it = m_account_code.find(addr);
655 if(it != m_account_code.end() && (it->second.second || !write)) {
656 return it->second.first;
657 }
658
659 auto elem_key = make_buffer(code_key{addr});
660 auto maybe_v = get_key(elem_key, write);
661 if(!maybe_v.has_value()) {
662 return std::nullopt;
663 }
664
665 m_accessed_addresses.insert(addr);
666 auto& v = maybe_v.value();
667 if(v.size() == 0) {
668 m_account_code[addr] = {std::nullopt, write};
669 return std::nullopt;

678 auto evm_host::get_key(const cbdc::buffer& key, bool write) const
679 -> std::optional<broker::value_type> {
680 const auto* log_str = "get_key";
681 auto res_prom
682 = std::promise<broker::interface::try_lock_return_type>();
683 auto res_fut = res_prom.get_future();
684
685 m_log->trace(m_ticket_number, log_str, key.to_hex(), "write =", write);
686
687 auto ret = m_try_lock_callback(
688 key,
689 write ? broker::lock_type::write : broker::lock_type::read,
690 [&](const broker::interface::try_lock_return_type& res) {
691 res_prom.set_value(res);
692 });
693
694 if(!ret) {
695 m_log->trace(m_ticket_number,
696 "failed to make try_lock request, retrying");
697 m_retry = true;
698 return std::nullopt;
699 }
700
701 for(size_t i = 0;; i++) {
702 auto status = res_fut.wait_for(std::chrono::seconds(1));
703 if(status == std::future_status::ready) {
704 break;
705 }
706 m_log->trace(m_ticket_number,
707 "still waits for",
708 key.to_hex(),
709 write,
710 i);
711 }
712
713 auto res = res_fut.get();
714
715 m_log->trace(m_ticket_number, "got key", key.to_hex());
716
717 return std::visit(
718 overloaded{
719 [&](broker::value_type& val) -> std::optional<cbdc::buffer> {
720 return val;
721 },
722 [&](broker::interface::error_code& /* err */)
723 -> std::optional<cbdc::buffer> {
724 m_retry = true;
725 return std::nullopt;
726 },
727 [&](runtime_locking_shard::shard_error& /* err */)
728 -> std::optional<cbdc::buffer> {
729 m_retry = true;
730 return std::nullopt;
731 },
732 },
733 res);
734 }
