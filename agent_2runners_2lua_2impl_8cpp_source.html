44 auto lua_runner::run() -> bool {
45 // TODO: use custom allocator to limit memory allocation
46 m_state
47 = std::shared_ptr<lua_State>(luaL_newstate(), [](lua_State* s) {
48 lua_close(s);
49 });
50
51 if(!m_state) {
52 m_log->error("Failed to allocate new lua state");
53 m_result_callback(error_code::internal_error);
54 return true;
55 }
56
57 // TODO: provide custom environment limited only to safe library
58 // methods
59 luaL_openlibs(m_state.get());
60
61 lua_register(m_state.get(), "check_sig", &lua_runner::check_sig);
62
63 static constexpr auto function_name = "contract";
64
65 auto load_ret = luaL_loadbufferx(m_state.get(),
66 m_function.c_str(),
67 m_function.size(),
68 function_name,
69 "b");
70 if(load_ret != LUA_OK) {
71 m_log->error("Failed to load function chunk");
72 m_result_callback(error_code::function_load);
73 return true;
74 }
75
76 if(lua_pushlstring(m_state.get(), m_param.c_str(), m_param.size())
77 == nullptr) {
78 m_log->error("Failed to push function params");
79 m_result_callback(error_code::internal_error);
80 return true;
81 }
82
83 schedule_contract();
84
85 return true;
86 }

88 void lua_runner::contract_epilogue(int n_results) {
89 if(n_results != 1) {
90 m_log->error("Contract returned more than one result");
91 m_result_callback(error_code::result_count);
92 return;
93 }
94
95 if(lua_istable(m_state.get(), -1) != 1) {
96 m_log->error("Contract did not return a table");
97 m_result_callback(error_code::result_type);
98 return;
99 }
100
101 auto results = runtime_locking_shard::state_update_type();
102
103 lua_pushnil(m_state.get());
104 while(lua_next(m_state.get(), -2) != 0) {
105 auto key_buf = get_stack_string(-2);
106 if(!key_buf.has_value()) {
107 m_log->error("Result key is not a string");
108 m_result_callback(error_code::result_key_type);
109 return;
110 }
111 auto value_buf = get_stack_string(-1);
112 if(!value_buf.has_value()) {
113 m_log->error("Result value is not a string");
114 m_result_callback(error_code::result_value_type);
115 return;
116 }
117
118 results.emplace(std::move(key_buf.value()),
119 std::move(value_buf.value()));
120
121 lua_pop(m_state.get(), 1);
122 }
123
124 m_log->trace(this, "running calling result callback");
125 m_result_callback(std::move(results));
126 m_log->trace(this, "lua_runner finished contract epilogue");
127 }

129 auto lua_runner::get_stack_string(int index) -> std::optional<buffer> {
130 if(lua_isstring(m_state.get(), index) != 1) {
131 return std::nullopt;
132 }
133 size_t sz{};
134 const auto* str = lua_tolstring(m_state.get(), index, &sz);
135 assert(str != nullptr);
136 auto buf = buffer();
137 buf.append(str, sz);
138 return buf;
139 }

141 void lua_runner::schedule_contract() {
142 int n_results{};
143 auto resume_ret = lua_resume(m_state.get(), nullptr, 1, &n_results);
144 if(resume_ret == LUA_YIELD) {
145 if(n_results != 1) {
146 m_log->error("Contract yielded more than one key");
147 m_result_callback(error_code::yield_count);
148 return;
149 }
150 auto key_buf = get_stack_string(-1);
151 if(!key_buf.has_value()) {
152 m_log->error("Contract did not yield a string");
153 m_result_callback(error_code::yield_type);
154 return;
155 }
156 lua_pop(m_state.get(), n_results);
157 auto success
158 = m_try_lock_callback(std::move(key_buf.value()),
159 broker::lock_type::write,
160 [&](auto res) {
161 handle_try_lock(std::move(res));
162 });
163 if(!success) {
164 m_log->error("Failed to issue try lock command");
165 m_result_callback(error_code::internal_error);
166 }
167 } else if(resume_ret != LUA_OK) {
168 const auto* err = lua_tostring(m_state.get(), -1);
169 m_log->error("Error running contract:", err);
170 m_result_callback(error_code::exec_error);
171 } else {
172 contract_epilogue(n_results);
173 }
174 }

176 void lua_runner::handle_try_lock(
177 const broker::interface::try_lock_return_type& res) {
178 auto maybe_error = std::visit(
179 overloaded{
180 [&](const broker::value_type& v) -> std::optional<error_code> {
181 if(lua_pushlstring(m_state.get(), v.c_str(), v.size())
182 == nullptr) {
183 m_log->error("Failed to push yield params");
184 return error_code::internal_error;
185 }
186 return std::nullopt;
187 },
188 [&](const broker::interface::error_code& /* e */)
189 -> std::optional<error_code> {
190 m_log->error("Broker error acquiring lock");
191 return error_code::lock_error;
192 },
193 [&](const runtime_locking_shard::shard_error& e)
194 -> std::optional<error_code> {
195 if(e.m_error_code
196 == runtime_locking_shard::error_code::wounded) {
197 return error_code::wounded;
198 }
199 m_log->error("Shard error acquiring lock");
200 return error_code::lock_error;
201 }},
202 res);
203 if(maybe_error.has_value()) {
204 m_result_callback(maybe_error.value());
205 return;
206 }
207 schedule_contract();
208 }

210 auto lua_runner::check_sig(lua_State* L) -> int {
211 int n = lua_gettop(L);
212 if(n != 3) {
213 lua_pushliteral(L, "not enough arguments");
214 lua_error(L);
215 }
216 for(int i = 1; i <= n; i++) {
217 if(lua_isstring(L, i) != 1) {
218 lua_pushliteral(L, "invalid argument");
219 lua_error(L);
220 }
221 }
222
223 size_t sz{};
224 const auto* str = lua_tolstring(L, 1, &sz);
225 assert(str != nullptr);
226 pubkey_t key{};
227 if(sz != key.size()) {
228 lua_pushliteral(L, "invalid pubkey");
229 lua_error(L);
230 }
231 std::memcpy(key.data(), str, sz);
232
233 str = lua_tolstring(L, 2, &sz);
234 assert(str != nullptr);
235 signature_t sig{};
236 if(sz != sig.size()) {
237 lua_pushliteral(L, "invalid signature");
238 lua_error(L);
239 }
240 std::memcpy(sig.data(), str, sz);
241
242 secp256k1_xonly_pubkey pubkey{};
243 if(secp256k1_xonly_pubkey_parse(secp_context.get(),
244 &pubkey,
245 key.data())
246 != 1) {
247 lua_pushliteral(L, "invalid pubkey");
248 lua_error(L);
249 }
250
251 str = lua_tolstring(L, 3, &sz);
252 assert(str != nullptr);
253 auto sha = CSHA256();
254 auto unsigned_str = std::vector<unsigned char>(sz);
255 std::memcpy(unsigned_str.data(), str, sz);
256 sha.Write(unsigned_str.data(), sz);
257 hash_t sighash{};
258 sha.Finalize(sighash.data());
259
260 if(secp256k1_schnorrsig_verify(secp_context.get(),
261 sig.data(),
262 sighash.data(),
263 &pubkey)
264 != 1) {
265 lua_pushliteral(L, "invalid signature");
266 lua_error(L);
267 }
268
269 return 0;
270 }
